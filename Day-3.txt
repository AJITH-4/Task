for...
let Array = [1,2,3,4,5];
for (i=1; i<=Array.length; i++){ 
    var n = Math.floor(i*2);
    console.log (n);
}

forin...
let Array = [1,2,3,4,5];
for(let index in Array){
console.log(index);
}

forof...
let Array = [1,2,3,4,5];
for(let value of Array){
console.log(value);
}

for each...
let sum = 0;
let numbers = [1,2,3,4,5];
numbers.forEach(myFunction);

function myFunction(item) {
  sum += item;
}
console.log(sum);
