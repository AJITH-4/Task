// Do the below programs in anonymous function & IIFE
// 1.Print odd numbers in an array
// Anonymous function

let add = function (array){
    oddsum = 0;
for(i=0;i<=array.length;i++){
    if(array[i]%2==0){
        oddsum+=array[i];
       
    }
}console.log(oddsum);
}
add([1,2,3,4,5,6]);

//IIEF 

(function add (array){
    oddsum = 0;
for(i=0;i<=array.length;i++){
    if(array[i]%2==0){
        oddsum+=array[i];
       
    }
}console.log(oddsum);
})([1,2,3,4,5,6]);
